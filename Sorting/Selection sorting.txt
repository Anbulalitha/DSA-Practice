int[] arr = {64, 25, 12, 22, 11};
for (int i = 0; i < arr.Length - 1; i++) {
    int minIdx = i;
    for (int j = i + 1; j < arr.Length; j++) {
        if (arr[j] < arr[minIdx]) minIdx = j;
    }
    int temp = arr[minIdx];
    arr[minIdx] = arr[i];
    arr[i] = temp;
}
Console.WriteLine("Sorted: " + string.Join(", ", arr));
